159  @Override
160  public synchronized Pushable push(Object o)
161  {
162  try
163  {
164  Queue<Object> q = m_inputQueues[m_index];
165  q.add(o);
166  }
167  catch (ArrayIndexOutOfBoundsException e)
168  {
169  throw new PushableException(e);
170  }
171  // Check if each input queue has an event ready
172  for (int i = 0; i < m_inputArity; i++)
173  {
174  Queue<Object> queue = m_inputQueues[i];
175  if (queue.isEmpty())
176  {
177  // One of them doesn't: we can't produce an output yet
178  return this;
179  }
180  }
181  // Pick an event from each input queue
182  Object[] inputs = new Object[m_inputArity];
183  for (int i = 0; i < m_inputArity; i++)
184  {
185  Queue<Object> queue = m_inputQueues[i];
186  Object ob = queue.remove();
187  inputs[i] = ob;
188  }
189  // Compute output event
190  m_tempQueue.clear();
191  boolean outs;
192  try
193  {
194  outs = compute(inputs, m_tempQueue);
195  }
196  catch (ProcessorException e)
197  {
198  throw new PushableException(e);
199  }
200  // put everything in outputEvent, because of duplicated code in notifyEndOfTrace
201  outputEvent(outs);
202  return this;
203  }

205  @Override
206  public void notifyEndOfTrace()
207  {
208  // Nothing to do if the Pushable has already been notified
209  if (m_hasBeenNotifiedOfEndOfTrace)
210  {
211  return;
212  }
213  m_hasBeenNotifiedOfEndOfTrace = true;
214 
215  m_tempQueue.clear();
216  boolean outs;
217  try
218  {
219  outs = onEndOfTrace(m_tempQueue);
220  }
221  catch (ProcessorException e)
222  {
223  throw new PushableException(e);
224  }
225  outputEvent(outs);
226 
227  // Notifies the output pushables of the end of the trace
228  for (int i = 0; i < m_outputPushables.length; i++)
229  {
230  Pushable p = m_outputPushables[i];
231  if (p == null)
232  {
233  throw new PushableException("Output " + i
234  + " of this processor is connected to nothing", getProcessor());
235  }
236  p.notifyEndOfTrace();

246  private final void outputEvent(boolean outs)
247  {
248  if (outs && !m_tempQueue.isEmpty())
249  {
250  for (Object[] evt : m_tempQueue)
251  {
252  if (evt != null)
253  {
254  for (int i = 0; i < m_outputPushables.length; i++)
255  {
256  Pushable p = m_outputPushables[i];
257  if (p == null)
258  {
259  throw new PushableException(
260  "Output " + i + " of processor " + SynchronousProcessor.this
261  + " is connected to nothing", getProcessor());
262  }
263  p.push(evt[i]);
264  }
265  }
266  }
267  m_tempQueue.clear();
268  }

357  @Override
358  public synchronized boolean hasNext()
359  {
360  Queue<Object> out_queue = m_outputQueues[m_index];
361  // If an event is already waiting in the output queue,
362  // return it and don't pull anything from the input
363  if (!out_queue.isEmpty())
364  {
365  return true;
366  }
367  // Check if each pullable has an event ready
368  for (int tries = 0; tries < Processor.MAX_PULL_RETRIES; tries++)
369  {
370  for (int i = 0; i < m_inputArity; i++)
371  {
372  Pullable p = m_inputPullables[i];
373  if (p == null)
374  {
375  throw new PullableException("Input " + i + " of processor " + SynchronousProcessor.this
376  + " is connected to nothing", getProcessor());
377  }
378  boolean status = p.hasNext();
379  if (!status)
380  {
381  if (m_hasBeenNotifiedOfEndOfTrace)
382  {
383  return false;
384  }
385  Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
386  boolean b = onEndOfTrace(last_queue);
387  m_hasBeenNotifiedOfEndOfTrace = true;
388  if (!b)
389  {
390  return false;
391  }
392  for (Object[] front : last_queue)
393  {
394  for (int j = 0; j < m_outputArity; j++)
395  {
396  m_outputQueues[j].add(front[j]);
397  }
398  }
399  return true;
400  }
401  }
402  // We are here only if every input pullable has answered YES
403  // Pull an event from each
404  Object[] inputs = new Object[m_inputArity];
405  for (int i = 0; i < m_inputArity; i++)
406  {
407  Pullable p = m_inputPullables[i];
408  // Don't check for p == null, we did it above
409  Object o = p.pull();
410  inputs[i] = o;
411  }
412  // Compute output event(s)
413  m_tempQueue.clear();
414  boolean computed;
415  try
416  {
417  computed = compute(inputs, m_tempQueue);
418  }
419  catch (ProcessorException e)
420  {
421  throw new PullableException(e);
422  }
423  NextStatus status_to_return = NextStatus.NO;
424  if (!computed)
425  {
426  // No output will ever be returned: stop there
427  return false;
428  }
429  if (!m_tempQueue.isEmpty())
430  {
431  // We computed an output event; add it to the output queue
432  // and answer YES
433  for (Object[] evt : m_tempQueue)
434  {
435  if (evt != null)
436  {
437  for (int i = 0; i < m_outputArity; i++)
438  {
439  Queue<Object> queue = m_outputQueues[i];
440  queue.add(evt[i]);
441  }
442  status_to_return = NextStatus.YES;
443  }
444  else
445  {
446  // This source will NEVER output anything again
447  m_tempQueue.clear();
448  return false;
449  }
450  }
451  if (status_to_return == NextStatus.YES)
452  {
453  m_tempQueue.clear();
454  return true;
455  }
456  }
457  // Otherwise, try the whole thing again
458  }
459  m_tempQueue.clear();
460  return false;
461  }

463  @Override
464  public synchronized NextStatus hasNextSoft()
465  {
466  Queue<Object> out_queue = m_outputQueues[m_index];
467  // If an event is already waiting in the output queue,
468  // return yes and don't pull anything from the input
469  if (!out_queue.isEmpty())
470  {
471  return NextStatus.YES;
472  }
473  // Check if each pullable has an event ready
474  for (int i = 0; i < m_inputArity; i++)
475  {
476  Pullable p = m_inputPullables[i];
477  NextStatus status = p.hasNextSoft();
478  if (status == NextStatus.NO)
479  {
480  if (m_hasBeenNotifiedOfEndOfTrace)
481  {
482  return NextStatus.NO;
483  }
484  Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
485  boolean b = onEndOfTrace(last_queue);
486  m_hasBeenNotifiedOfEndOfTrace = true;
487  if (!b)
488  {
489  return NextStatus.NO;
490  }
491  for (Object[] front : last_queue)
492  {
493  for (int j = 0; j < m_outputArity; j++)
494  {
495  m_outputQueues[j].add(front[j]);
496  }
497  }
498  return NextStatus.YES;
499  }
500  if (status == NextStatus.MAYBE)
501  {
502  return NextStatus.MAYBE;
503  }
504  }
505  // We are here only if every input pullable has answered YES
506  // Pull an event from each
507  Object[] inputs = new Object[m_inputArity];
508  {
509  int i = 0;
510  for (Pullable p : m_inputPullables)
511  {
512  inputs[i] = p.pullSoft();
513  i++;
514  }
515  }
516  // Compute output event(s)
517  NextStatus status_to_return = NextStatus.MAYBE;
518  m_tempQueue.clear();
519  boolean computed;
520  try
521  {
522  computed = compute(inputs, m_tempQueue);
523  }
524  catch (ProcessorException e)
525  {
526  throw new PullableException(e);
527  }
528  if (!computed)
529  {
530  status_to_return = NextStatus.NO;
531  }
532  else if (!m_tempQueue.isEmpty())
533  {
534  for (Object[] evt : m_tempQueue)
535  {
536  if (evt != null)
537  {
538  // We computed an output event; add it to the output queue
539  // and answer YES
540  int i = 0;
541  for (Queue<Object> queue : m_outputQueues)
542  {
543  queue.add(evt[i]);
544  i++;
545  }
546  status_to_return = NextStatus.YES;
547  }
548  }
549  }
550  return status_to_return;
551  }
