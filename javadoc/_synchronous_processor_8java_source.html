151 @Override
152 public Pushable push(Object o)
153 {
154 try
155 {
156 Queue<Object> q = m_inputQueues[m_index];
157 q.add(o);
158 }
159 catch (ArrayIndexOutOfBoundsException e)
160 {
161 throw new PushableException(e);
162 }
163 // Check if each input queue has an event ready
164 for (int i = 0; i < m_inputArity; i++)
165 {
166 Queue<Object> queue = m_inputQueues[i];
167 if (queue.isEmpty())
168 {
169 // One of them doesn't: we can't produce an output yet
170 return this;
171 }
172 }
173 // Pick an event from each input queue
174 Object[] inputs = new Object[m_inputArity];
175 for (int i = 0; i < m_inputArity; i++)
176 {
177 Queue<Object> queue = m_inputQueues[i];
178 Object ob = queue.remove();
179 inputs[i] = ob;
180 }
181 // Compute output event
182 m_tempQueue.clear();
183 boolean outs;
184 try
185 {
186 outs = compute(inputs, m_tempQueue);
187 }
188 catch (ProcessorException e)
189 {
190 throw new PushableException(e);
191 }
192 // put everything in outputEvent, because of duplicated code in notifyEndOfTrace
193 outputEvent(outs);
194 return this;
195 }
196
197 @Override
198 public void notifyEndOfTrace()
199 {
200 m_hasBeenNotifiedOfEndOfTrace[m_index] = true;
201 if (m_notifiedEndOfTraceDownstream || !allNotifiedEndOfTrace())
202 {
203 return;
204 }
205 m_notifiedEndOfTraceDownstream = true;
206 m_tempQueue.clear();
207 boolean outs;
208 try
209 {
210 outs = onEndOfTrace(m_tempQueue);
211 }
212 catch (ProcessorException e)
213 {
214 throw new PushableException(e);
215 }
216 outputEvent(outs);
217
218 // Notifies the output pushables of the end of the trace
219 for (int i = 0; i < m_outputPushables.length; i++)
220 {
221 Pushable p = m_outputPushables[i];
222 if (p == null)
223 {
224 throw new PushableException("Output " + i
225 + " of this processor is connected to nothing", getProcessor());
226 }
227 p.notifyEndOfTrace();
228 }
229 }
230
237 private final void outputEvent(boolean outs)
238 {
239 if (!m_tempQueue.isEmpty())
240 {
241 for (Object[] evt : m_tempQueue)
242 {
243 if (evt != null)
244 {
245 for (int i = 0; i < m_outputPushables.length; i++)
246 {
247 Pushable p = m_outputPushables[i];
248 if (p == null)
249 {
250 throw new PushableException(
251 "Output " + i + " of processor " + SynchronousProcessor.this
252 + " is connected to nothing", getProcessor());
253 }
254 p.push(evt[i]);
255 }
256 }
257 }
258 m_tempQueue.clear();
259 }
260 if (!outs)
261 {
262 for (int i = 0; i < m_outputPushables.length; i++)
263 {
264 m_outputPushables[i].notifyEndOfTrace();
265 }
266 }
267 }

301 @Override
302 public void remove()
303 {
304 // Cannot remove an event on a pullable
305 throw new UnsupportedOperationException();

308 @Override
309 public Object pullSoft()
310 {
311 if (hasNextSoft() != NextStatus.YES)
312 {
313 return null;
314 }
315 Queue<Object> out_queue = m_outputQueues[m_index];
316 // If an event is already waiting in the output queue,
317 // return it and don't pull anything from the input
318 if (!out_queue.isEmpty())
319 {
320 return out_queue.remove();
321 }
322 return null;
323 }
324
325 @Override
326 public Object pull()
327 {
328 if (!hasNext())
329 {
330 throw new NoSuchElementException();
331 }
332 Queue<Object> out_queue = m_outputQueues[m_index];
333 // If an event is already waiting in the output queue,
334 // return it and don't pull anything from the input
335 if (!out_queue.isEmpty())
336 {
337 return out_queue.remove();
338 }
339 throw new NoSuchElementException();
340 }
341
342 @Override
343 @SuppressWarnings("squid:S2272") // since() pull throws the exception
344 public final Object next()
345 {
346 return pull();
347 }

349 @Override
350 public boolean hasNext()
351 {
352 Queue<Object> out_queue = m_outputQueues[m_index];
353 // If an event is already waiting in the output queue,
354 // return it and don't pull anything from the input
355 if (!out_queue.isEmpty())
356 {
357 return true;
358 }
359 // Check if each pullable has an event ready
360 for (int tries = 0; tries < Processor.MAX_PULL_RETRIES; tries++)
361 {
362 for (int i = 0; i < m_inputArity; i++)
363 {
364 Pullable p = m_inputPullables[i];
365 if (p == null)
366 {
367 throw new PullableException("Input " + i + " of processor " + SynchronousProcessor.this
368 + " is connected to nothing", getProcessor());
369 }
370 boolean status = p.hasNext();
371 if (!status)
372 {
373 if (allNotifiedEndOfTrace())
374 {
375 return false;
376 }
377 Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
378 boolean b = onEndOfTrace(last_queue);
379 m_hasBeenNotifiedOfEndOfTrace[i] = true;
380 if (!b)
381 {
382 return false;
383 }
384 for (Object[] front : last_queue)
385 {
386 for (int j = 0; j < m_outputArity; j++)
387 {
388 m_outputQueues[j].add(front[j]);
389 }
390 }
391 return true;
392 }
393 }
394 // We are here only if every input pullable has answered YES
395 // Pull an event from each
396 Object[] inputs = new Object[m_inputArity];
397 for (int i = 0; i < m_inputArity; i++)
398 {
399 Pullable p = m_inputPullables[i];
400 // Don't check for p == null, we did it above
401 Object o = p.pull();
402 inputs[i] = o;
403 }
404 // Compute output event(s)
405 m_tempQueue.clear();
406 boolean computed;
407 try
408 {
409 computed = compute(inputs, m_tempQueue);
410 }
411 catch (ProcessorException e)
412 {
413 throw new PullableException(e);
414 }
415 NextStatus status_to_return = NextStatus.NO;
416 if (!computed && m_tempQueue.isEmpty())
417 {
418 // No output will ever be returned: stop there
419 return false;
420 }
421 if (!m_tempQueue.isEmpty())
422 {
423 // We computed an output event; add it to the output queue
424 // and answer YES
425 for (Object[] evt : m_tempQueue)
426 {
427 if (evt != null)
428 {
429 for (int i = 0; i < m_outputArity; i++)
430 {
431 Queue<Object> queue = m_outputQueues[i];
432 queue.add(evt[i]);
433 }
434 status_to_return = NextStatus.YES;
435 }
436 else
437 {
438 // This source will NEVER output anything again
439 m_tempQueue.clear();
440 return false;
441 }
442 }
443 if (status_to_return == NextStatus.YES)
444 {
445 m_tempQueue.clear();
446 return true;
447 }
448 }
449 // Otherwise, try the whole thing again
450 }
451 m_tempQueue.clear();
452 return false;
453 }
454
455 @Override
456 public NextStatus hasNextSoft()
457 {
458 Queue<Object> out_queue = m_outputQueues[m_index];
459 // If an event is already waiting in the output queue,
460 // return yes and don't pull anything from the input
461 if (!out_queue.isEmpty())
462 {
463 return NextStatus.YES;
464 }
465 // Check if each pullable has an event ready
466 for (int i = 0; i < m_inputArity; i++)
467 {
468 Pullable p = m_inputPullables[i];
469 NextStatus status = p.hasNextSoft();
470 if (status == NextStatus.NO)
471 {
472 if (allNotifiedEndOfTrace())
473 {
474 return NextStatus.NO;
475 }
476 Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
477 boolean b = onEndOfTrace(last_queue);
478 m_hasBeenNotifiedOfEndOfTrace[i] = true;
479 if (!b)
480 {
481 return NextStatus.NO;
482 }
483 for (Object[] front : last_queue)
484 {
485 for (int j = 0; j < m_outputArity; j++)
486 {
487 m_outputQueues[j].add(front[j]);
488 }
489 }
490 return NextStatus.YES;
491 }
492 if (status == NextStatus.MAYBE)
493 {
494 return NextStatus.MAYBE;
495 }
496 }
497 // We are here only if every input pullable has answered YES
498 // Pull an event from each
499 Object[] inputs = new Object[m_inputArity];
500 {
501 int i = 0;
502 for (Pullable p : m_inputPullables)
503 {
504 inputs[i] = p.pullSoft();
505 i++;
506 }
507 }
508 // Compute output event(s)
509 NextStatus status_to_return = NextStatus.MAYBE;
510 m_tempQueue.clear();
511 boolean computed;
512 try
513 {
514 computed = compute(inputs, m_tempQueue);
515 }
516 catch (ProcessorException e)
517 {
518 throw new PullableException(e);
519 }
520 if (!computed)
521 {
522 status_to_return = NextStatus.NO;
523 }
524 else if (!m_tempQueue.isEmpty())
525 {
526 for (Object[] evt : m_tempQueue)
527 {
528 if (evt != null)
529 {
530 // We computed an output event; add it to the output queue
531 // and answer YES
532 int i = 0;
533 for (Queue<Object> queue : m_outputQueues)
534 {
535 queue.add(evt[i]);
536 i++;
537 }
538 status_to_return = NextStatus.YES;
539 }
540 }
541 }
542 return status_to_return;
543 }
