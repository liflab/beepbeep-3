130 @Override
131 public Pushable push(Object o)
132 {
133 Object[] out_front = new Object[m_outputPullables.length];
134 if (m_frontNumber[m_index] < m_numCurrentFront)
135 {
136 // This is an event from a front that has already
137 // been processed. Don't bother putting it in the
138 // queue or evaluate the function.
139 m_frontNumber[m_index]++;
140 return this;
141 }
142 assert m_frontNumber[m_index] >= m_numCurrentFront;
143 m_inputQueues[m_index].add(o);
144 boolean evaluated = true;
145 // the following is actually a bounded while(evaluated)
146 for (int loop_count = 0; loop_count < MAX_LOOPS && evaluated; loop_count++)
147 {
148 for (int i = 0; i < m_inputPullables.length; i++)
149 {
150 if (m_frontNumber[i] == m_numCurrentFront)
151 {
152 if (!m_inputQueues[i].isEmpty())
153 {
154 m_inputFront[i] = m_inputQueues[i].remove();
155 m_frontNumber[i]++;
156 }
157 else
158 {
159 m_inputFront[i] = null;
160 }
161 }
162 }
163 evaluated = m_function.evaluatePartial(m_inputFront, out_front, m_context);
164 if (evaluated)
165 {
166 for (int i = 0; i < out_front.length; i++)
167 {
168 m_outputPushables[i].push(out_front[i]);
169 }
170 for (int i = 0; i < m_inputFront.length; i++)
171 {
172 m_inputFront[i] = null;
173 }
174 m_numCurrentFront++;
175 }
176 }
177 return this;
178 }
