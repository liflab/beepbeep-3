128  @Override
129  public Pushable push(Object o)
130  {
131  Object[] out_front = new Object[m_outputPullables.length];
132  if (m_frontNumber[m_index] < m_numCurrentFront)
133  {
134  // This is an event from a front that has already
135  // been processed. Don't bother putting it in the
136  // queue or evaluate the function.
137  m_frontNumber[m_index]++;
138  return this;
139  }
140  assert m_frontNumber[m_index] >= m_numCurrentFront;
141  m_inputQueues[m_index].add(o);
142  boolean evaluated = true;
143  // the following is actually a bounded while(evaluated)
144  for (int loop_count = 0; loop_count < MAX_LOOPS && evaluated; loop_count++)
145  {
146  for (int i = 0; i < m_inputPullables.length; i++)
147  {
148  if (m_frontNumber[i] == m_numCurrentFront)
149  {
150  if (!m_inputQueues[i].isEmpty())
151  {
152  m_inputFront[i] = m_inputQueues[i].remove();
153  m_frontNumber[i]++;
154  }
155  else
156  {
157  m_inputFront[i] = null;
158  }
159  }
160  }
161  evaluated = m_function.evaluatePartial(m_inputFront, out_front, m_context);
162  if (evaluated)
163  {
164  for (int i = 0; i < out_front.length; i++)
165  {
166  m_outputPushables[i].push(out_front[i]);
167  }
168  for (int i = 0; i < m_inputFront.length; i++)
169  {
170  m_inputFront[i] = null;
171  }
172  m_numCurrentFront++;
173  }
174  }
175  return this;
176  }
