325  public synchronized Map<Integer, Processor> cloneInto(GroupProcessor group, boolean with_state)
326  {
327  super.duplicateInto(group);
328  group.m_notifySources = m_notifySources;
329  Map<Integer, Processor> new_procs = new HashMap<Integer, Processor>();
330  Processor start = null;
331  // Clone every processor of the original group
332  for (Processor p : m_processors)
333  {
334  if (start == null && p.getOutputArity() > 0)
335  {
336  start = p;
337  }
338  Processor clone_p = copyProcessor(p, with_state);
339  new_procs.put(p.getId(), clone_p);
340  group.addProcessor(clone_p);
341  }
342  // Re-pipe the inputs and outputs like in the original group
343  associateEndpoints(group, new_procs);
344  // Re-pipe the internal processors like in the original group
345  CopyCrawler cc = new CopyCrawler(new_procs);
346  cc.crawl(start);
347  return new_procs;
348  }

359  protected synchronized void associateEndpoints(GroupProcessor group,
360  Map<Integer, Processor> new_procs)
361  {
362  // Re-pipe the inputs like in the original group
363  for (Map.Entry<Integer, ProcessorAssociation> entry : m_inputPullableAssociations.entrySet())
364  {
365  int input_number = entry.getKey();
366  ProcessorAssociation pa = entry.getValue();
367  Processor clone_p = new_procs.get(pa.m_processor.getId());
368  group.associateInput(input_number, clone_p, pa.m_ioNumber);
369  }
370  // Re-pipe the outputs like in the original group
371  for (Map.Entry<Integer, ProcessorAssociation> entry : m_outputPushableAssociations.entrySet())
372  {
373  int output_number = entry.getKey();
374  ProcessorAssociation pa = entry.getValue();
375  Processor clone_p = new_procs.get(pa.m_processor.getId());
376  group.associateOutput(output_number, clone_p, pa.m_ioNumber);
377  }
378  }

391  protected static synchronized Processor copyProcessor(Processor p, boolean with_state)
392  {
393  Processor clone_p = p.duplicate(with_state);
394  clone_p.setContext(p.m_context);
395  if (with_state)
396  {
397  // Put same content in input and output queues
398  for (int i = 0; i < p.m_inputQueues.length; i++)
399  {
400  clone_p.m_inputQueues[i].addAll(p.m_inputQueues[i]);
401  }
402  for (int i = 0; i < p.m_outputQueues.length; i++)
403  {
404  clone_p.m_outputQueues[i].addAll(p.m_outputQueues[i]);
405  }
406  }
407  return clone_p;
408  }

450  @Override
451  public synchronized void visit(Processor p)
452  {
453  int out_arity = p.getOutputArity();
454  for (int i = 0; i < out_arity; i++)
455  {
456  Pushable push = p.getPushableOutput(i);
457  if (push != null)
458  {
459  Processor target = push.getProcessor();
460  int j = push.getPosition();
461  Processor new_p;
462  Processor new_target;
463  synchronized (m_correspondences)
464  {
465  new_p = m_correspondences.get(p.getId());
466  new_target = m_correspondences.get(target.getId());
467  }
468  if (new_p != null && new_target != null)
469  {
470  // new_p and new_target may be null if they refer to a processor
471  // outside of the group
472  Connector.connect(new_p, i, new_target, j);
473  }
474  }
475  }
476  }
477  }
