439 public Map<Integer, Processor> cloneInto(GroupProcessor group, boolean with_state)
440 {
441 super.duplicateInto(group);
442 if (group.m_eventTracker != null)
443 {
444 group.m_eventTracker.add(group);
445 }
446 group.m_notifySources = m_notifySources;
447 Map<Integer, Processor> new_procs = new HashMap<Integer, Processor>();
448 EventTracker new_tracker = null;
449 if (m_innerTracker != null)
450 {
451 new_tracker = m_innerTracker.getCopy(false);
452 }
453 group.m_innerTracker = new_tracker;
454 Processor start = null;
455 // Clone every processor of the original group
456 for (Processor p : m_processors)
457 {
458 if (start == null && p.getOutputArity() > 0)
459 {
460 start = p;
461 }
462 Processor clone_p = copyProcessor(p, with_state);
463 new_procs.put(p.getId(), clone_p);
464 group.addProcessor(clone_p);
465 }
466 // Re-pipe the inputs and outputs like in the original group
467 associateEndpoints(group, new_procs);
468 // Re-pipe the internal processors like in the original group
469 CopyCrawler cc = new CopyCrawler(new_procs, new_tracker);
470 cc.crawl(start);
471 return new_procs;
472 }

483 protected void associateEndpoints(GroupProcessor group,
484 Map<Integer, Processor> new_procs)
485 {
486 // Re-pipe the inputs like in the original group
487 for (Map.Entry<Integer, ProcessorAssociation> entry : m_inputPullableAssociations.entrySet())
488 {
489 int input_number = entry.getKey();
490 ProcessorAssociation pa = entry.getValue();
491 Processor clone_p = new_procs.get(pa.m_processor.getId());
492 group.associateInput(input_number, clone_p, pa.m_ioNumber);
493 }
494 // Re-pipe the outputs like in the original group
495 for (Map.Entry<Integer, ProcessorAssociation> entry : m_outputPushableAssociations.entrySet())
496 {
497 int output_number = entry.getKey();
498 ProcessorAssociation pa = entry.getValue();
499 Processor clone_p = new_procs.get(pa.m_processor.getId());
500 group.associateOutput(output_number, clone_p, pa.m_ioNumber);
501 }

604 protected static Processor copyProcessor(Processor p, boolean with_state)
605 {
606 Processor clone_p = p.duplicate(with_state);
607 clone_p.setContext(p.m_context);
608 if (with_state)
609 {
610 // Put same content in input and output queues
611 for (int i = 0; i < p.m_inputQueues.length; i++)
612 {
613 clone_p.m_inputQueues[i].addAll(p.m_inputQueues[i]);
614 }
615 for (int i = 0; i < p.m_outputQueues.length; i++)
616 {
617 clone_p.m_outputQueues[i].addAll(p.m_outputQueues[i]);
618 }
619 }
620 return clone_p;
621 }

641 protected static class CopyCrawler extends PipeCrawler
642 {
643 private final Map<Integer, Processor> m_correspondences;
644
645 private final EventTracker m_tracker;
646
647 public CopyCrawler(Map<Integer, Processor> correspondences, EventTracker tracker)
648 {
649 super();
650 m_correspondences = new HashMap<Integer, Processor>();
651 m_correspondences.putAll(correspondences);
652 m_tracker = tracker;
653 }
654
655 @Override
656 public void crawl(Processor start)
657 {
658 if (start.getOutputArity() == 0)
659 {
660 throw new UnsupportedOperationException(
661 "A copy crawl cannot start from a processor of output arity 0.");
662 }
663 super.crawl(start);
664 }
665
666 @Override
667 public void visit(Processor p)
668 {
669 int out_arity = p.getOutputArity();
670 for (int i = 0; i < out_arity; i++)
671 {
672 Pushable push = p.getPushableOutput(i);
673 if (push != null)
674 {
675 Processor target = push.getProcessor();
676 int j = push.getPosition();
677 Processor new_p;
678 Processor new_target;
679 new_p = m_correspondences.get(p.getId());
680 new_target = m_correspondences.get(target.getId());
681 if (new_p != null && new_target != null)
682 {
683 // new_p and new_target may be null if they refer to a processor
684 // outside of the group
685 Connector.connect(m_tracker, new_p, i, new_target, j);
686 }
687 }
688 }

865 @Override
866 public void notifyEndOfTrace() throws PushableException
867 {
868 m_hasBeenNotifiedOfEndOfTrace[m_position] = true;
869 if (!allNotifiedEndOfTrace())
870 {
871 return;
872 }
873 // Notify the end of trace on all the inner Pushables
874 for (Pushable p : m_inputPushables)
875 {
876 ((ProxyPushable) p).m_pushable.notifyEndOfTrace();
877 }
878
879 // Collect from processor the events to generate for the end
880 Queue<Object[]> temp_queue = new ArrayDeque<Object[]>();
881 boolean outs;
882 try
883 {
884 outs = onEndOfTrace(temp_queue);
885 }
886 catch (ProcessorException e)
887 {
888 throw new PushableException(e);
889 }
890 outputEvent(temp_queue, outs);
891
892 // Notify the output pushables of the end of the trace
893 for (int i = 0; i < m_outputPushables.length; i++)
894 {
895 ProcessorAssociation pa = m_outputPushableAssociations.get(i);
896 Pushable p = pa.m_processor.getPushableOutput(pa.m_ioNumber);
897 if (p == null)
898 {
899 throw new PushableException("Output " + i
900 + " of this processor is connected to nothing", getProcessor());
901 }
902 p.notifyEndOfTrace();
903 }
904 }

913 private final void outputEvent(Queue<Object[]> temp_queue, boolean outs)
914 {
915 if (outs && !temp_queue.isEmpty())
916 {
917 for (Object[] evt : temp_queue)
918 {
919 if (evt != null)
920 {
921 for (int i = 0; i < m_outputPushables.length; i++)
922 {
923 Pushable p = m_outputPushables[i];
924 if (p == null)
925 {
926 throw new PushableException(
927 "Output " + i + " of this processor is connected to nothing", getProcessor());
928 }
929 p.push(evt[i]);
930 }
931 }
932 }

1112 @SuppressWarnings("unchecked")
1113 @Override
1114 public GroupProcessor readState(Object o) throws ProcessorException
1115 {
1116 Map<String,Object> contents = (HashMap<String,Object>) o;
1117 // Create processor
1118 GroupProcessor gp = new GroupProcessor(((Number) contents.get("in-arity")).intValue(),
1119 ((Number) contents.get("out-arity")).intValue());
1120 // Add internal processors (regular and source)
1121 Map<Integer,Processor> procs = new HashMap<Integer,Processor>();
1122 for (Processor p : (List<Processor>) contents.get("processors"))
1123 {
1124 gp.addProcessor(p);
1125 procs.put(p.getId(), p);
1126 }
1127 for (Processor p : (Set<Processor>) contents.get("sources"))
1128 {
1129 gp.addProcessor(p);
1130 procs.put(p.getId(), p);
1131 }
1132 // Connect each of them according to the connections
1133 Set<Connector.Connection> connections = (Set<Connector.Connection>) contents.get("connections");
1134 for (Connector.Connection conn : connections)
1135 {
1136 Processor p1 = procs.get(conn.m_sourceProcessorId);
1137 Processor p2 = procs.get(conn.m_destinationProcessorId);
1138 if (p1 == null || p2 == null)
1139 {
1140 // Ignore
1141 continue;
1142 }
1143 Connector.connect(p1, conn.m_sourcePipeNumber, p2, conn.m_destinationPipeNumber);
1144 }
1145 // Associate group's inputs to processor inputs
1146 for (List<Integer> list : ((Set<List<Integer>>) contents.get("input-associations")))
1147 {
1148 int io_1 = list.get(0);
1149 int io_2 = list.get(1);
1150 int p_id = list.get(2);
1151 if (!procs.containsKey(p_id))
1152 {
1153 // Ignore
1154 continue;
1155 }
1156 Processor p = procs.get(p_id);
1157 gp.associateInput(io_1, p, io_2);
1158 }
1159 // Associate group's outputs to processor outputs
1160 for (List<Integer> list : ((Set<List<Integer>>) contents.get("output-associations")))
1161 {
1162 int io_1 = list.get(0);
1163 int io_2 = list.get(1);
1164 int p_id = list.get(2);
1165 if (!procs.containsKey(p_id))
1166 {
1167 // Ignore
1168 continue;
1169 }
1170 Processor p = procs.get(p_id);
1171 gp.associateOutput(io_1, p, io_2);
1172 }
1173 gp.m_notifySources = (Boolean) contents.get("notify-sources");
1174 return gp;
1175 }
