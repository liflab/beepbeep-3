117 for (int index = m_processorIndex; index < m_processors.length; index++)
118 {
119 boolean has_null = false;
120 for (int i = 0; i < m_pullables.length; i++)

123 if (!status)
124 {
125 // No use in trying further
126 has_null = true;
127 break;
128 }
129 Object o = m_pullables[i].pull();
130 if (o == null)
131 {
132 has_null = true;
133 break;
134 }
135 out[i] = o;
136 }
137 if (!has_null)
138 {
139 outputs.add(out);
140 return true;
141 }
142 m_processorIndex++;
143 if (m_processorIndex < m_processors.length)
144 {
145 setPullablesTo(m_processorIndex);
146 }
147 }
