162 @Override
163 @SuppressWarnings("squid:S1168")
164 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
165 {
166 int size = m_events.size();
167 if (m_index >= size)
168 {
169 return false;
170 }
171 Object[] output = new Object[getOutputArity()];
172 Object event = m_events.get(m_index);
173 if (m_loop)
174 {
175 m_index = (m_index + 1) % size;
176 }
177 else
178 {
179 // If we don't loop, play the events only once
180 m_index++;
181 }
182 if (m_index > size && !m_loop)
183 {
184 // No more events from this queue
185 return false;
186 }
187 for (int i = 0; i < getOutputArity(); i++)
188 {
189 if (event == null)
190 {
191 // If one of the elements is null, don't output anything
192 return true;
193 }
194 output[i] = event;
195 }
196 outputs.add(output);
197 if (m_eventTracker != null)
198 {
199 int index = m_index - 1;
200 if (index < 0)
201 {
202 index += size;
203 }
204 for (int i = 0; i < getOutputArity(); i++)
205 {
206 associateTo(new QueueFunction(getId(), index), i, m_outputCount);
207 }
208 }
209 m_outputCount++;
210 return true;
211 }
