67  try
68  {

71  char[] char_array = new char[m_chunkSize];
72  int chars_read = m_br.read(char_array, 0, m_chunkSize);
73  // When the input is a pipe and we read the special character,
74  // this indicates the end of transmission
75  if (chars_read > 0)
76  {
77  char[] new_array = Arrays.copyOf(char_array, chars_read);
78  String st = new String(new_array);
79  if (!m_isFile && st.endsWith(END_CHARACTER))
80  {
81  // We don't read a file, but the input stream
82  // has the EOT character: trim this EOT form the output
83  st = st.substring(0, st.length() - 1);
84  // And remember this stream is over
85  m_hasReadEot = true;
86  }
87  outputs.add(new Object[] { st });
88  return true;
89  }
90  }
91  else
92  {
93  // If the underlying input source is not a pipe, the
94  // fact that the input stream is not ready means there
95  // is no more data to read.
96  if (m_isFile)
97  {
98  return false;
99  }
100  }
101  // At this point, we haven't read bytes, but we don't know if we'll
102  // be able to read some in the future: return true just in case
103  return true;
