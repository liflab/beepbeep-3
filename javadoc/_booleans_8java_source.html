446 public static boolean parseBoolValue(Object o)
447 {
448 if (o instanceof Boolean)
449 {
450 return (Boolean) o;
451 }
452 else if (o instanceof String)
453 {
454 String s = (String) o;
455 return s.compareToIgnoreCase("true") == 0 || s.compareToIgnoreCase("T") == 0
456 || s.compareToIgnoreCase("1") == 0;
457 }
458 if (o instanceof Number)
459 {
460 Number n = (Number) o;
461 return Math.abs(n.doubleValue()) >= 0.00001;
462 }
463 // When in doubt, return false
464 return false;
