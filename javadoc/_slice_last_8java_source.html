135 @SuppressWarnings("unchecked")
136 @Override
137 public SliceLast readState(Object o) throws ProcessorException
138 {
139 Map<String,Object> contents = (HashMap<String,Object>) o;
140 Function cleaning_function = (Function) contents.get("cleaning-function");
141 Function slicing_function = (Function) contents.get("slicing-function");
142 Processor proc = (Processor) contents.get("processor");
143 SliceLast s = new SliceLast(slicing_function, proc, cleaning_function);
144 s.m_sinks = (HashMap<Object,QueueSink>) contents.get("sinks");
145 s.m_slices = (HashMap<Object,Processor>) contents.get("slices");
146 s.m_currentList = (List<Object>) contents.get("list");
147 s.m_explodeArrays = (Boolean) contents.get("explode-arrays");
148 // Connect slices to sinks
149 for (Map.Entry<Object,Processor> entry : s.m_slices.entrySet())
150 {
151 Processor p = entry.getValue();
152 if (!s.m_sinks.containsKey(entry.getKey()))
153 {
154 throw new ProcessorException("Cannot restore the state of the slice processor");
155 }
156 QueueSink qs = s.m_sinks.get(entry.getKey());
157 Connector.connect(p, qs);
158 }
159 return s;
160 }
