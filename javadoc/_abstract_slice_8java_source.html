151 @Override
152 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
153 {
154 //System.out.println(inputs[0]);
155 int output_arity = getOutputArity();
156 Object[] f_value = new Object[1];
157 try
158 {
159 m_slicingFunction.evaluate(inputs, f_value);
160 }
161 catch (FunctionException e)
162 {
163 throw new ProcessorException(e);
164 }
165 Object slice_ids = f_value[0];
166 if (slice_ids == null)
167 {
168 // This event applies to no slice; don't bother processing it
169 return produceReturn(outputs);
170 }
171 Object[] slice_vals;
172 if (m_explodeArrays)
173 {
174 if (slice_ids.getClass().isArray())
175 {
176 slice_vals = (Object[]) slice_ids;
177 }
178 if (slice_ids instanceof Collection)
179 {
180 Collection<?> col = (Collection<?>) slice_ids;
181 slice_vals = new Object[col.size()];
182 int i = 0;
183 for (Object o : col)
184 {
185 slice_vals[i] = o;
186 i++;
187 }
188 }
189 else
190 {
191 slice_vals = new Object[] { slice_ids };
192 }
193 }
194 else
195 {
196 slice_vals = new Object[] { slice_ids };
197 }
198 for (Object slice_id : slice_vals)
199 {
200 Set<Object> slices_to_process = new HashSet<Object>();
201 if (slice_id instanceof ToAllSlices || slice_id == null)
202 {
203 slices_to_process.addAll(m_slices.keySet());
204 }
205 else
206 {
207 if (!m_slices.containsKey(slice_id))
208 {
209 // First time we see this value: create new slice
210 Processor p = m_processor.duplicate();
211 m_slices.put(slice_id, p);
212 addContextFromSlice(p, slice_id);
213 QueueSink sink = new QueueSink(output_arity);
214 Connector.connect(p, sink);
215 m_sinks.put(slice_id, sink);
216 if (m_eventTracker != null)
217 {
218 //p.setEventTracker(m_eventTracker.getCopy(false));
219 m_sliceIndices.put(slice_id, new ArrayList<Integer>());
220 }
221 }
222 slices_to_process.add(slice_id);
223 }
224 for (Object s_id : slices_to_process)
225 {
226 // Find processor corresponding to that slice
227 Processor slice_p = m_slices.get(s_id);
228 // If this slice hasn't been cleaned up...
229 if (slice_p != null)
230 {
231 QueueSink sink_p = m_sinks.get(s_id);
232 if (m_eventTracker != null)
233 {
234 m_sliceIndices.get(s_id).add(m_inputCount);
235 }
236 // Push the input into the processor
237 for (int i = 0; i < inputs.length; i++)
238 {
239 Object o_i = inputs[i];
240 Pushable p = slice_p.getPushableInput(i);
241 p.push(o_i);
242 }
243 // Collect the output from that processor
244 Object[] out = sink_p.remove();
245 // Can we clean that slice?
246 Object[] can_clean = new Object[1];
247 if (m_cleaningFunction != null)
248 {
249 try
250 {
251 m_cleaningFunction.evaluate(out, can_clean);
252 }
253 catch (FunctionException e)
254 {
255 throw new ProcessorException(e);
256 }
257 }
258 if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
259 && (Boolean) (can_clean[0]))
260 {
261 // Yes: remove the processor for that slice
262 m_slices.remove(s_id);
263 m_sinks.remove(s_id);
264 }
265 handleNewSliceValue(s_id, out[0], outputs);
266 }
267 }
268 }
269 m_inputCount++;
270 return produceReturn(outputs);
271 }

332 protected void copyInto(AbstractSlice as, boolean with_state)
333 {
334 // We do not duplicate cleaning function as it is taken care of by the children's constructor
335 as.m_explodeArrays = m_explodeArrays;
336 as.m_processor = m_processor.duplicate(with_state);
337 as.m_slicingFunction = m_slicingFunction.duplicate(with_state);
338 as.setContext(m_context);
339 for (Map.Entry<Object,Processor> e : m_slices.entrySet())
340 {
341 Processor p_dup = e.getValue().duplicate(with_state);
342 QueueSink qs = new QueueSink();
343 Connector.connect(p_dup, qs);
344 as.m_slices.put(e.getKey(), p_dup);
345 as.m_sinks.put(e.getKey(), qs);
346 }
347 for (Map.Entry<Object,List<Integer>> e : m_sliceIndices.entrySet())
348 {
349 List<Integer> list = new ArrayList<Integer>();
350 list.addAll(e.getValue());
351 as.m_sliceIndices.put(e.getKey(), list);
352 }

355 @Override
356 protected boolean onEndOfTrace(Queue<Object[]> outputs) throws ProcessorException
357 {
358 boolean new_events = false;
359 // Send the message to all slices
360 for (Map.Entry<Object, Processor> e : m_slices.entrySet())
361 {
362 Processor p = e.getValue();
363 Pushable pp = p.getPushableInput(0);
364 QueueSink sink = m_sinks.get(e.getKey());
365 int size_before = sink.getQueue().size();
366 pp.notifyEndOfTrace();
367 int size_after = sink.getQueue().size();
368 if (size_after > size_before)
369 {
370 new_events = true;
371 handleNewSliceValue(e.getKey(), sink.getQueue().peek(), outputs);
372 }
373 }
374 if (new_events)
375 {
376 return produceReturn(outputs);
377 }
378 return false;
379 }
