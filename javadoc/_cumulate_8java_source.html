49  protected boolean compute(Object[] inputs, Object[] outputs)
50  {
51  // We override compute() from FunctionProcessor, only to complete the
52  // association between input and output events (each output event depends on
53  // the current input event and also the previous output event front)
54  boolean b = super.compute(inputs, outputs);
55  if (m_eventTracker != null)
56  {
57  for (int j = 0; j < outputs.length; j++)
58  {
59  if (m_outputCount > 1)
60  {
61  for (int k = 0; k < outputs.length; k++)
62  {
63  // The output event is associated to the current input event
64  // (this is already taken care of by super.compute)...
65  // ...and to the previous output event
66  // -1 and -2 since the count has already been incremented by the
67  // call to super.compute() above
68  associateToOutput(j, m_outputCount - 2, k, m_outputCount - 1);
69 
70  }
71  }
72  else
73  {
74  for (int k = 0; k < outputs.length; k++)
75  {
76  associateTo(new StartValue(getId(), j), k, m_outputCount - 1);
77  }
78  }
79  }
80  }
81  return b;
82  }
