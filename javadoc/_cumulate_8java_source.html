46  @Override
47  protected boolean compute(Object[] inputs, Object[] outputs)
48  {
49  // We override compute() from FunctionProcessor, only to complete the
50  // association between input and output events (each output event depends on
51  // the current input event and also the previous output event front)
52  boolean b = super.compute(inputs, outputs);
53  if (m_eventTracker != null)
54  {
55  for (int j = 0; j < outputs.length; j++)
56  {
57  if (m_outputCount > 1)
58  {
59  for (int k = 0; k < outputs.length; k++)
60  {
61  // The output event is associated to the current input event
62  // (this is already taken care of by super.compute)...
63  // ...and to the previous output event
64  // -1 and -2 since the count has already been incremented by the
65  // call to super.compute() above
66  associateToOutput(j, m_outputCount - 2, k, m_outputCount - 1);
67 
68  }
69  }
70  else
71  {
72  for (int k = 0; k < outputs.length; k++)
73  {
74  associateTo(new StartValue(getId(), j), k, m_outputCount - 1);
75  }
76  }
77  }
78  }
79  return b;
80  }
