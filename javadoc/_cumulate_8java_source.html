66 @Override
67 protected boolean compute(Object[] inputs, Object[] outputs)
68 {
69 // We override compute() from ApplyFunction, only to complete the
70 // association between input and output events (each output event depends on
71 // the current input event and also the previous output event front)
72 boolean b = super.compute(inputs, outputs);
73 if (m_eventTracker != null)
74 {
75 for (int j = 0; j < outputs.length; j++)
76 {
77 if (m_outputCount > 1)
78 {
79 for (int k = 0; k < outputs.length; k++)
80 {
81 // The output event is associated to the current input event
82 // (this is already taken care of by super.compute)...
83 // ...and to the previous output event
84 // -1 and -2 since the count has already been incremented by the
85 // call to super.compute() above
86 associateToOutput(j, m_outputCount - 2, k, m_outputCount - 1);
87
88 }
89 }
90 else
91 {
92 for (int k = 0; k < outputs.length; k++)
93 {
94 associateTo(new StartValue(getId(), j), k, m_outputCount - 1);
95 }
96 }
97 }
98 }
99 return b;
100 }
