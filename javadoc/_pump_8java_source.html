79 @Override
80 public void run()
81 {
82 m_run = true;
83 Pullable pullable = getPullableInput(0);
84 Pushable pushable = getPushableOutput(0);
85 while (m_run && pullable.hasNext())
86 {
87 Object o = pullable.pull();
88 pushable.push(o);
89 if (m_interval >= 0)
90 {
91 try
92 {
93 Thread.sleep(m_interval);
94 }
95 catch (InterruptedException e)
96 {
97 // Restore interrupted state
98 Thread.currentThread().interrupt();
99 }
100 }
101 }
102 pushable.notifyEndOfTrace();
103 }
