73 public Multiset addAll(Multiset b)
74 {
75 for (Object o : b.keySet())

300 @Override
301 public boolean removeAll(Collection<?> arg0)
302 {
303 boolean removed = false;
304 if (arg0 instanceof Multiset)
305 {
306 Multiset set = (Multiset) arg0;
307 for (Object o : set.keySet())
308 {
309 // Look out: the call must be on the LHS. Otherwise,
310 // if removed == true, the call would never be evaluated.
311 removed = remove(o, set.get(o)) || removed;
312 }
313 }
314 else
315 {
316 // Normal collection
317 for (Object o : arg0)
318 {
319 removed = remove(o, 1) || removed;
320 }
321 }
322 return removed;
323 }

325 @Override
326 public boolean retainAll(Collection<?> arg0)
327 {
328 boolean changed = false;
329 if (arg0 instanceof Multiset)
330 {
331 Multiset set = (Multiset) arg0;
332 for (Object o : set.keySet())
333 {
334 int target_card = set.get(o);
335 if (m_map.get(o) > target_card)
336 {
337 changed = true;
338 m_map.put(o, target_card);
339 }
340 }
341 for (Object o : m_map.keySet())
342 {
343 if (!set.contains(o))
344 {
345 m_map.remove(o);
346 changed = true;
347 }
348 }
349 }
350 else
351 {
352 // Normal set
353 for (Object o : arg0)
354 {
355 if (get(o) > 1)
356 {
357 m_map.put(o, 1);
358 changed = true;
359 }
360 }
361 for (Object o : m_map.keySet())
362 {
363 if (!arg0.contains(o))
364 {
365 m_map.remove(o);
366 changed = true;
367 }
368 }
369 }
370 return changed;
