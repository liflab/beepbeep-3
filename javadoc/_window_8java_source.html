69 @Override
70 public void reset()
71 {
72 super.reset();
73 int arity = getInputArity();
74 m_window = new LinkedList[arity];
75 m_innerInputs = new Pushable[arity];
76 m_processor.reset();
77 for (int i = 0; i < arity; i++)
78 {
79 m_window[i] = new LinkedList<Object>();
80 m_innerInputs[i] = m_processor.getPushableInput(i);
81 }
82 m_sink.reset();
83 Connector.connect(m_processor, m_sink);
84 }

86 @Override
87 @SuppressWarnings("squid:S3516")
88 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
89 {
90 // Add the inputs to each window
91 m_inputCount++;
92 boolean windows_ok = true;
93 int arity = inputs.length;
94 for (int i = 0; i < arity; i++)
95 {
96 LinkedList<Object> q = m_window[i];
97 q.add(inputs[i]);
98 int size_diff = q.size() - m_width;
99 leftTrim(size_diff, q);
100 if (size_diff < 0)
101 {
102 // Window is still to small to compute
103 windows_ok = false;
104 }
105 }
106 Object[] out = null;
107 if (windows_ok) // All windows have the proper width
108 {
109 m_processor.reset();
110 m_sink.reset();
111 int input_arity = getInputArity();
112 for (int i = 0; i < m_width; i++)
113 {
114 for (int j = 0; j < input_arity; j++)
115 {
116 // Feed
117 LinkedList<Object> q = m_window[j];
118 Object o = q.get(i);
119 Pushable p = m_innerInputs[j];
120 p.push(o);
121 p.notifyEndOfTrace();
122 }
123 out = m_sink.getLast();
124 }
125 }
126 if (out == null)
127 {
128 // Don't return false, otherwise it would signal that no
129 // event will every be produced in the future
130 return true;
131 }
132 else
133 {
134 m_outputCount++;
135 if (m_eventTracker != null)
136 {
137 for (int i = 1 - m_width; i <= 0; i++)
138 {
139 associateToInput(0, m_inputCount + i, 0, m_outputCount);
140 }
141 }
142 }
143 outputs.add(out);
144 return true;
145 }

156 protected static void leftTrim(int n, List<Object> q)
157 {
158 if (n <= 0)
159 {
160 return;
161 }
162 if (n >= q.size())
163 {
164 q.clear();
165 return;
166 }
167 for (int i = 0; i < n; i++)
168 {
169 q.remove(0);
170 }
171 }
