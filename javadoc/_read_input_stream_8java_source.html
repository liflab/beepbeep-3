137 @Override
138 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
139 {
140 if (m_hasReadEot)
141 {
142 // We received EOT previously: no more output to produce
143 return false;
144 }
145 try
146 {
147 if (m_br.ready())
148 {
149 char[] char_array = new char[m_chunkSize];
150 int chars_read = m_br.read(char_array, 0, m_chunkSize);
151 // When the input is a pipe and we read the special character,
152 // this indicates the end of transmission
153 if (chars_read > 0)
154 {
155 char[] new_array = Arrays.copyOf(char_array, chars_read);
156 char[] out_array;
157 if (!m_isFile && new_array[new_array.length - 1] == END_CHARACTER)
158 {
159 // We don't read a file, but the input stream
160 // has the EOT character: trim this EOT form the output
161 out_array = Arrays.copyOf(new_array, chars_read - 1);
162 // And remember this stream is over
163 m_hasReadEot = true;
164 m_br.close();
165 }
166 else
167 {
168 out_array = new_array;
169 }
170 outputs.add(new Object[] { out_array });
171 return !m_hasReadEot;
172 }
173 }
174 else
175 {
176 // If the underlying input source is not a pipe, the
177 // fact that the input stream is not ready means there
178 // is no more data to read.
179 if (m_isFile)
180 {
181 return false;
182 }
183 }
184 // At this point, we haven't read bytes, but we don't know if we'll
185 // be able to read some in the future: return true just in case
186 return true;
187 }
188 catch (IOException e)
189 {
190 throw new ProcessorException(e);
191 }
192 }

246 @Override
247 public boolean hasNext()
248 {
249 if (m_isFile)
250 {
251 return super.hasNext();
252 }
253 // If named pipe, wait until hasNextSoft has
254 // answered either YES or NO; otherwise loop forever
255 Queue<Object> out_queue = m_outputQueues[getPosition()];
256 // If an event is already waiting in the output queue,
257 // return it and don't pull anything from the input
258 if (!out_queue.isEmpty())
259 {
260 return true;
261 }
262 // Next line is different from OutputPullable, which has a bounded for loop
263 while (true)
264 {
265 for (int i = 0; i < m_inputArity; i++)
266 {
267 Pullable p = m_inputPullables[i];
268 if (p == null)
269 {
270 throw new PullableException("Input " + i + " of processor " + ReadInputStream.this
271 + " is connected to nothing", getProcessor());
272 }
273 boolean status = p.hasNext();
274 if (!status)
275 {
276 if (allNotifiedEndOfTrace())
277 {
278 return false;
279 }
280 Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
281 boolean b = onEndOfTrace(last_queue);
282 m_hasBeenNotifiedOfEndOfTrace[i] = true;
283 if (!b)
284 {
285 return false;
286 }
287 for (Object[] front : last_queue)
288 {
289 for (int j = 0; j < m_outputArity; j++)
290 {
291 m_outputQueues[j].add(front[j]);
292 }
293 }
294 return true;
295 }
296 }
297 // We are here only if every input pullable has answered YES
298 // Pull an event from each
299 Object[] inputs = new Object[m_inputArity];
300 for (int i = 0; i < m_inputArity; i++)
301 {
302 Pullable p = m_inputPullables[i];
303 // Don't check for p == null, we did it above
304 Object o = p.pull();
305 inputs[i] = o;
306 }
307 // Compute output event(s)
308 m_tempQueue.clear();
309 boolean computed;
310 try
311 {
312 computed = compute(inputs, m_tempQueue);
313 }
314 catch (ProcessorException e)
315 {
316 throw new PullableException(e);
317 }
318 NextStatus status_to_return = NextStatus.NO;
319 if (!computed && m_tempQueue.isEmpty())
320 {
321 // No output will ever be returned: stop there
322 return false;
323 }
324 if (!m_tempQueue.isEmpty())
325 {
326 // We computed an output event; add it to the output queue
327 // and answer YES
328 for (Object[] evt : m_tempQueue)
329 {
330 if (evt != null)
331 {
332 for (int i = 0; i < m_outputArity; i++)
333 {
334 Queue<Object> queue = m_outputQueues[i];
335 queue.add(evt[i]);
336 }
337 status_to_return = NextStatus.YES;
338 }
339 else
340 {
341 // This source will NEVER output anything again
342 m_tempQueue.clear();
343 return false;
344 }
345 }
346 if (status_to_return == NextStatus.YES)
347 {
348 m_tempQueue.clear();
349 return true;
350 }
351 }
352 // Otherwise, try the whole thing again
353 }
354 }
