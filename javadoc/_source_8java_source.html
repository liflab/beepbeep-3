51 public final void push()
52 {
53 Queue<Object[]> output = new ArrayDeque<Object[]>(1);
54 try
55 {
56 compute(null, output);
57 }
58 catch (ProcessorException e)
59 {
60 throw new PushableException(e);
61 }
62 if (output.isEmpty())
63 {
64 return;
65 }
66 for (Object[] evt : output)
67 {
68 if (evt != null && !allNull(evt))
69 {
70 for (int i = 0; i < output.size(); i++)
71 {
72 Pushable p = m_outputPushables[i];
73 p.push(evt[i]);
74 }
75 }
76 }
