342 protected class Timer implements Runnable
343 {
344 protected volatile boolean m_run = true;
345
346 public void stop()
347 {
348 m_run = false;
349 }
350
351 @Override
352 public void run()
353 {
354 m_run = true;
355 while (m_run)
356 {
357 try
358 {
359 Thread.sleep(m_outputInterval);
360 }
361 catch (InterruptedException e)
362 {
363 // Restore interrupted state
364 Thread.currentThread().interrupt();
365 }
366 Pushable p = getPushableOutput(0);
367 m_lock.lock();
368 p.push(m_packedEvents);
369 m_packedEvents = new LinkedList<Object>();
370 m_lock.unlock();
371 }
372 }
373 }

376 public Pullable getPullableOutput(int position)
377 {
378 return new Pullable.PullNotSupported(this, position);
379 }

540
541 @Override
542 public int compare(T arg0, T arg1)
543 {
544 Object[] out1 = new Object[1];
545 Object[] out2 = new Object[1];
546 f.evaluate(new Object[] {arg0}, out1);
547 f.evaluate(new Object[] {arg1}, out2);
548 return attemptComparison(out1[0], out2[0]);
549 }};
550 }
551
552 @SuppressWarnings("unchecked")
553 @Override
554 public List getValue(Collection c)
555 {
556 List<T> list = new ArrayList<T>();
557 list.addAll(c);
558 Collections.sort(list, m_comparator);
559 return list;
560 }

562 protected static int attemptComparison(Object o1, Object o2)
563 {
564 if (o1 instanceof Number && o2 instanceof Number)
565 {
566 double d1 = ((Number) o1).doubleValue();
567 double d2 = ((Number) o2).doubleValue();
568 if (d1 == d2)
569 {
570 return 0;
571 }
572 if (d1 < d2)
573 {
574 return -1;
575 }
576 return 1;
577 }
578 if (o1 instanceof String && o2 instanceof String)
579 {
580 return ((String) o1).compareTo((String) o2);
581 }
582 return 0;
583 }
