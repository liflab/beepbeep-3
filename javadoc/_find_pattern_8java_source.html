97 @Override
98 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
99 {
100 m_contents += (String) inputs[0];
101 Matcher mat = m_pattern.matcher(m_contents);
102 int last_end = 0;
103 while (mat.find())
104 {
105 String matched;
106 if (mat.groupCount() > 0)
107 {
108 matched = mat.group(1);
109 if (matched == null)
110 {
111 throw new Error("Capturing group 1 of " + m_pattern + "did not match in " + m_contents);
112 }
113 } else {
114 matched = mat.group();
115 }
116 if (m_trim)
117 {
118 matched = matched.trim();
119 }
120 outputs.add(new Object[] { matched });
121 last_end = mat.end();
122 }
123 m_contents = m_contents.substring(last_end);
124 return true;
