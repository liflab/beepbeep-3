195 public static class RunOn extends SynchronousProcessor
196 {
197 protected Processor m_processor;
198
199 protected transient SinkLast m_sink;
200
201 protected transient Pushable m_pushable;
202
203 protected final Object[] m_default;
204
210 public RunOn(Processor processor)
211 {
212 super(1, processor.getOutputArity());
213 int out_arity = processor.getOutputArity();
214 m_processor = processor;
215 m_pushable = m_processor.getPushableInput();
216 m_sink = new SinkLast(out_arity);
217 Connector.connect(m_processor, m_sink);
218 m_default = null;
219 }
220
226 public RunOn(Processor processor, Object[] default_values)
227 {
228 super(1, processor.getOutputArity());
229 int out_arity = processor.getOutputArity();
230 m_processor = processor;
231 m_pushable = m_processor.getPushableInput();
232 m_sink = new SinkLast(out_arity);
233 Connector.connect(m_processor, m_sink);
234 m_default = default_values;
235 }
236
237 @Override
238 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
239 {
240 boolean empty_input = false;
241 m_processor.reset();
242 m_sink.reset();
243 if (inputs[0].getClass().isArray())
244 {
245 empty_input = ((Object[]) inputs[0]).length == 0;
246 for (Object o : (Object[]) inputs[0])
247 {
248 m_pushable.push(o);
249 }
250 }
251 else
252 {
253 empty_input = ((Collection<?>) inputs[0]).size() == 0;
254 for (Object o : (Collection<?>) inputs[0])
255 {
256 m_pushable.push(o);
257 }
258 }
259 Object[] last = m_sink.getLast();
260 if (last != null)
261 {
262 Object[] outs = new Object[last.length];
263 for (int i = 0; i < last.length; i++)
264 {
265 outs[i] = last[i];
266 }
267 outputs.add(outs);
268 }
269 else if (empty_input && m_default != null)
270 {
271 outputs.add(m_default);
272 }
273 return true;
274 }
275
276 @Override
277 public RunOn duplicate(boolean with_state)
278 {
279 return new RunOn(m_processor.duplicate(with_state), m_default);
280 }
