347 public static class MergeMaps extends UniformProcessor
348 {
352 protected Map<Object,Set<Object>> m_map;
353
357 public MergeMaps()
358 {
359 super(1, 1);
360 m_map = new HashMap<Object,Set<Object>>();
361 }
362
363 @Override
364 protected boolean compute(Object[] inputs, Object[] outputs)
365 {
366 Map<?,?> m = (Map<?,?>) inputs[0];
367 for (Object m_o : m.entrySet())
368 {
369 if (m_o instanceof Map.Entry)
370 {
371 Map.Entry<?,?> e = (Map.Entry<?,?>) m_o;
372 Object key = e.getKey();
373 Object value = e.getValue();
374 if (value == null)
375 {
376 outputs[0] = m_map;
377 return true;
378 }
379 Set<Object> s_value;
380 if (m_map.containsKey(key))
381 {
382 s_value = m_map.get(key);
383 }
384 else
385 {
386 s_value = new HashSet<Object>();
387 }
388 if (value instanceof Collection)
389 {
390 s_value.addAll((Collection<?>) value);
391 }
392 else
393 {
394 s_value.add(value);
395 }
396 m_map.put(key, s_value);
397 }
398 }
399 outputs[0] = m_map;
400 return true;
401 }
402
403 @Override
404 public void reset()
405 {
406 super.reset();
407 m_map.clear();
408 }
409
410 @Override
411 public MergeMaps duplicate(boolean with_state)
412 {
413 MergeMaps mm = new MergeMaps();
414 if (with_state)
415 {
416 for (Map.Entry<Object,Set<Object>> e : m_map.entrySet())
417 {
418 HashSet<Object> new_set = new HashSet<Object>();
419 new_set.addAll(e.getValue());
420 mm.m_map.put(e.getKey(), new_set);
421 }
422 }
423 return mm;
424 }
