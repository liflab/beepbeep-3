170  @Override
171  public NextStatus hasNextSoft()
172  {
173  if (!m_outputQueues[0].isEmpty())
174  {
175  return NextStatus.YES;
176  }
177  boolean all_no = true;
178  NextStatus out = NextStatus.MAYBE;
179  for (Pullable p : m_inputPullables)
180  {
181  NextStatus ns = p.hasNextSoft();
182  if (ns != NextStatus.NO)
183  {
184  all_no = false;
185  }
186  if (ns == NextStatus.YES)
187  {
188  // We don't do a "break" here.
189  // We must go through ALL pullables, even if we encounter one
190  // that says yes. Otherwise, we might end up pulling events from
191  // the same pullable all the time.
192  out = NextStatus.YES;
193  }
194  }
195  if (all_no)
196  {
197  return NextStatus.NO;
198  }
199  return out;
200  }

202  @Override
203  public boolean hasNext()
204  {
205  if (!m_outputQueues[0].isEmpty())
206  {
207  return true;
208  }
209  boolean all_no = true;
210  NextStatus out = NextStatus.MAYBE;
211  for (int i = 0; i < Processor.MAX_PULL_RETRIES; i++)
212  {
213  for (Pullable p : m_inputPullables)
214  {
215  boolean ns = p.hasNext();
216  if (ns)
217  {
218  all_no = false;
219  // We don't do a "break" here.
220  // We must go through ALL pullables, even if we encounter one
221  // that says yes. Otherwise, we might end up pulling events from
222  // the same pullable all the time.
223  out = NextStatus.YES;
224  }
225  }
226  if (all_no)
227  {
228  return false;
229  }
230  if (out == NextStatus.YES)
231  {
232  return true;
233  }
234  }

236  // anything; declare defeat and return NO
