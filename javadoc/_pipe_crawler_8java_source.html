47 public void crawl(Processor start)
48 {
49 Queue<Processor> to_visit = new LinkedList<Processor>();
50 Set<Processor> visited = new HashSet<Processor>();
51 to_visit.add(start);
52 /*
53 * This is actually while(!to_visit.isEmpty()), whose number of iterations is
54 * bounded by s_maxCrawls for safety.
55 */
56 for (int num_crawls = 0; num_crawls < s_maxCrawls && !to_visit.isEmpty(); num_crawls++)
57 {
58 Processor proc = to_visit.remove();
59 visit(proc);
60 visited.add(proc);
61 int in_arity = proc.getInputArity();
62 int out_arity = proc.getOutputArity();
63 for (int i = 0; i < out_arity; i++)
64 {
65 Pushable p = proc.getPushableOutput(i);
66 if (p == null)
67 {
68 continue;
69 }
70 Processor target = p.getProcessor();
71 if (!to_visit.contains(target) && !visited.contains(target))
72 {
73 to_visit.add(target);
74 }
75 }
76 for (int i = 0; i < in_arity; i++)
77 {
78 Pullable p = proc.getPullableInput(i);
79 if (p == null)
80 {
81 continue;
82 }
83 Processor target = p.getProcessor();
84 if (!to_visit.contains(target) && !visited.contains(target))
85 {
86 to_visit.add(target);
87 }
88 }
89 }
90 }
