59 @Override
60 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
61 {
62 if (m_inputCount >= getDelay())
63 {
64 outputs.add(inputs);
65 if (m_eventTracker != null)
66 {
67 for (int i = 0; i < inputs.length; i++)
68 {
69 m_eventTracker.associateToInput(getId(), i, m_inputCount, i, m_outputCount);
70 }
71 }
72 m_outputCount++;
73 }
74 m_inputCount++;
75 return true;
76 }
77
78 @Override
79 public Trim duplicate(boolean with_state)
80 {

82 if (with_state)
83 {
84 t.m_inputCount = m_inputCount;

86 }
87 return t;
88 }
