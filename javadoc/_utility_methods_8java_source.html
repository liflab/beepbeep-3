130 public static String print(Object o)
131 {
132 if (o == null)
133 {
134 return "null";
135 }
136 if (o instanceof Collection<?>)
137 {
138 String out = "[";
139 for (Object e : (Collection<?>) o)
140 {
141 out += print(e);
142 }
143 out += "]";
144 return out;
145 }
146 if (o.getClass().isArray())
147 {
148 Object[] a = (Object[]) o;
149 String out = "[";
150 for (int i = 0; i < a.length; i++)
151 {
152 if (i > 0)
153 {
154 out += ",";
155 }
156 out += print(a[i]);
157 }
158 out += "]";
159 return out;
160 }
161 return o.toString();
162 }
