200  protected void execute() throws IOException
201  {
202  Process process = null;
203  try
204  {
205  ProcessBuilder builder = new ProcessBuilder(m_command);
206  process = builder.start();
207  m_stderrGobbler = new StreamGobbler(process.getErrorStream(), "ERR");
208  m_stdoutGobbler = new StreamGobbler(process.getInputStream(), "IN");
209  // Send data into stdin of process
210  if (m_stdin != null)
211  {
212  OutputStream process_stdin = process.getOutputStream();
213  process_stdin.write(m_stdin, 0, m_stdin.length);
214  process_stdin.flush();
215  process_stdin.close();
216  }
217  // Start gobblers
218  m_stderrGobbler.start();
219  m_stdoutGobbler.start();
220  m_errorCode = process.waitFor();
221  do
222  {
223  // Wait for both gobblers to finish
224  } while (!m_stop && (m_stderrGobbler.isAlive() || m_stdoutGobbler.isAlive()));
225  }
226  catch (InterruptedException e)
