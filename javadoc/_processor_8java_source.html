839 @SuppressWarnings("unchecked")
840 @Override
841 public final Processor read(ObjectReader<?> reader, Object o) throws ProcessorException
842 {
843 Map<String, Object> contents = null;
844 try
845 {
846 contents = (Map<String,Object>) reader.read(o);
847 }
848 catch (ReadException e)
849 {
850 throw new ProcessorException(e);
851 }
852 Processor p = null;
853 if (contents.containsKey("contents"))
854 {
855 Object o_contents = contents.get("contents");
856 try
857 {
858 p = readState(o_contents);
859 }
860 catch (UnsupportedOperationException e)
861 {
862 throw new ProcessorException(e);
863 }
864 }
865 if (p == null)
866 {
867 throw new ProcessorException("The processor returned null with being deserialized");
868 }
869 p.m_inputCount = ((Number) contents.get("input-count")).intValue();
870 p.m_outputCount = ((Number) contents.get("output-count")).intValue();
871 try
872 {
873 reader.setField(p, "m_uniqueId", ((Number) contents.get("id")).intValue());
874 }
875 catch (ReadException e)
876 {
877 throw new ProcessorException(e);
878 }
879 List<Queue<Object>> in_queues = (List<Queue<Object>>) contents.get("input-queues");
880 for (int i = 0; i < in_queues.size(); i++)
881 {
882 p.m_inputQueues[i] = in_queues.get(i);
883 }
884 List<Queue<Object>> out_queues = (List<Queue<Object>>) contents.get("output-queues");
885 for (int i = 0; i < out_queues.size(); i++)
886 {
887 p.m_outputQueues[i] = in_queues.get(i);
888 }
889 return p;
890 }
