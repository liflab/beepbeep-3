58 public static class BuildString extends UniformProcessor
59 {
63 protected StringBuilder m_builder;
64
68 protected String m_separator;
69
73 public BuildString()
74 {
75 super(1, 1);
76 m_builder = new StringBuilder();
77 m_separator = "";
78 }
79
80 public BuildString setSeparator(String separator)
81 {
82 m_separator = separator;
83 return this;
84 }
85
86 @Override
87 public void reset()
88 {
89 super.reset();
90 m_builder = new StringBuilder();
91 m_inputCount = 0;
92 }
93
94 @Override
95 protected boolean compute(Object[] inputs, Object[] outputs)
96 {
97 if (m_inputCount > 0)
98 {
99 m_builder.append(m_separator);
100 }
101 m_builder.append(inputs[0].toString());
102 outputs[0] = m_builder.toString();
103 m_inputCount++;
104 return true;
105 }
106
107 @Override
108 public Processor duplicate(boolean with_state)
109 {
110 BuildString bs = new BuildString();
111 bs.m_separator = m_separator;
112 if (with_state)
113 {
114 bs.m_builder.append(m_builder.toString());
115 bs.m_inputCount = m_inputCount;
116 }
117 return bs;
118 }

219 public static class SplitString extends UnaryFunction<String, Object>
220 {
224 public static final transient SplitString instance = new SplitString(":");
225
229 protected String m_separator = ":";
230
234 protected boolean m_trim = true;
235
236 public SplitString(String separator)
237 {
238 super(String.class, Object.class);
239 m_separator = separator;
240 }
241
249 public SplitString trim(boolean b)
250 {
251 m_trim = b;
252 return this;
253 }
254
255 @Override
256 public Object getValue(String s)
257 {
258 String[] parts = s.split(m_separator);
259 Object[] typed_parts = new Object[parts.length];
260 for (int i = 0; i < parts.length; i++)
261 {
262 if (m_trim)
263 {
264 typed_parts[i] = createConstantFromString(parts[i].trim());
265 }
266 else
267 {
268 typed_parts[i] = createConstantFromString(parts[i]);
269 }
270 }
271 return typed_parts;
272 }

497 {
498 for (int i = 0; i < Math.min(m_from.length, m_to.length); i++)
