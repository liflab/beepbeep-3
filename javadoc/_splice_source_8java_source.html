43 public SpliceSource(boolean read_stdin, String ... filenames)
44 {
45 super(1);
46 m_sources = new Processor[filenames.length + (read_stdin ? 1 : 0)];
47 for (int i = 0; i < filenames.length; i++)
48 {
49 m_sources[i] = getSource(filenames[i]);
50 }
51 if (read_stdin)
52 {
53 m_sources[filenames.length] = getSource("-");
54 }
55 m_streamIndex = 0;
56 }

58 public SpliceSource(String ... filenames)

63 @Override
64 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
65 {
66 Object o = getNextEvent();
67 if (o == null)
68 {
69 return false;
70 }
71 outputs.add(new Object[] {o});
72 return true;
73 }

75 @Override
76 public Processor duplicate(boolean with_state)

78 throw new UnsupportedOperationException("This source cannot be duplicated");
79 }

81 @Override
82 public void start()
83 {
84 super.start();
85 Pushable p = getPushableOutput(0);
86 Object o;
87 do
88 {
89 o = getNextEvent();
90 if (o != null)
91 {
92 p.push(o);
93 }
94 } while (o != null);
95 p.notifyEndOfTrace();
96 stop();
97 }
98
99 @Override
100 public void stop()
101 {
102 super.stop();
103 for (Processor p : m_sources)
104 {
105 p.stop();
106 }
107 }
108
109 protected Object getNextEvent()
110 {
111 while (m_streamIndex < m_sources.length)
112 {
113 Pullable p = m_sources[m_streamIndex].getPullableOutput();
114 if (p.hasNext())
115 {
116 return p.pull();
117 }
118 else
119 {
120 m_sources[m_streamIndex].stop();
121 m_streamIndex++;
122 }
123 }
124 return null;
125 }
