81 public static boolean isEqualTo(Object x, Object y)
82 {
83 if ((x == null) != (y == null))
84 {
85 return false;
86 }
87 if (x == null)
88 {
89 return true;
90 }
91 if (x instanceof Constant)
92 {
93 x = ((Constant) x).getValue();
94 }
95 if (y instanceof Constant)
96 {
97 y = ((Constant) y).getValue();
98 }
99 if (x instanceof MathList || x instanceof MathMap || x instanceof MathSet)
100 {
101 return x.equals(y);
102 }
103 if (x instanceof Collection && y instanceof Collection)
104 {
105 Collection<?> set_x = (Collection<?>) x;
106 Collection<?> set_y = (Collection<?>) y;
107 return set_x.size() == set_y.size() && set_x.containsAll(set_y);
108 }
109 if (x instanceof String && y instanceof String)
110 {
111 return ((String) x).compareTo((String) y) == 0;
112 }
113 if (x instanceof Number && y instanceof Number)
114 {
115 return ((Number) x).floatValue() == ((Number) y).floatValue();
116 }
117 return x.equals(y);
118 }
