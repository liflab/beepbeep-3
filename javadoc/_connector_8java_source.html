130 public static Processor connect(EventTracker tracker, Processor p1, int i, Processor p2, int j)
131 {
132 // First check for type compatibility
133 if (s_checkForTypes)
134 {
135 // This will throw an exception if the connection is impossible
136 checkForException(p1, i, p2, j);
137 }
138 if (p1 == p2)
139 {
140 // This is weird: you try to connect a processor to itself
141 throw new SelfLoopException(p1, i, p2, j);
142 }
143 // Pull
144 try
145 {
146 Pullable p1_out = p1.getPullableOutput(i);
147 p2.setPullableInput(j, p1_out);
148 }
149 catch (ArrayIndexOutOfBoundsException e)
150 {
151 throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
152 }
153 catch (UnsupportedOperationException e)
154 {
155 // It's OK. Some processors deliberately throw this
156 // exception to warn an end user that they don't have a pushable
157 // or a pullable, but the connector does not care.
158 }
159 // Push
160 try
161 {
162 Pushable p2_in = p2.getPushableInput(j);
163 p1.setPushableOutput(i, p2_in);
164 }
165 catch (ArrayIndexOutOfBoundsException e)
166 {
167 throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
168 }
169 catch (UnsupportedOperationException e)
170 {
171 // Same as above
172 }
173 if (tracker != null)
174 {
175 tracker.setConnection(p1.getId(), i, p2.getId(), j);
176 }
177 return p2;
178 }

226 public static Processor connect(EventTracker tracker, Processor ... procs)
227 {
228 if (procs.length == 1)
229 {
230 // If given only one processor, do nothing
231 return procs[0];
232 }
233 for (int j = 0; j < procs.length - 1; j++)
234 {
235 Processor p1 = procs[j];
236 Processor p2 = procs[j + 1];
237 int arity = p1.getOutputArity();
238 for (int i = 0; i < arity; i++)
239 {
240 connect(tracker, p1, i, p2, i);
241 }
242 }
243 return procs[procs.length - 1];
244 }

288 @SuppressWarnings("unused")
289 protected static void checkForException(Processor p1, int i, Processor p2, int j)
290 {
291 try
292 {
293 Class<?> out_class = p1.getOutputType(i);
294 if (s_checkForBounds && out_class == null)
295 {
296 // p1 has no output, so how would you connect it to p2?
297 throw new IndexOutOfBoundsException(p1, i, p2, j);
298 }
299 if (out_class != null && out_class.equals(Variant.class))
300 {
301 // Skip type checking
302 return;
303 }
304 /*@ non_null @*/ Set<Class<?>> in_classes = p2.getInputType(j);
305 if (in_classes.isEmpty())
306 {
307 if (s_checkForBounds)
308 {
309 // p2 has no input, so how would you connect it to p1?
310 throw new IndexOutOfBoundsException(p1, i, p2, j);
311 }
312 else
313 {
314 // We don't mind that we connect an output to something
315 // that has no input
316 return;
317 }
318 }
319 for (Class<?> in_class : in_classes)
320 {
321 if (in_class.equals(Variant.class) || in_class.isAssignableFrom(out_class)
322 || in_class.equals(Object.class))
323 {
324 // Found a compatible in/out pair of types: return without exception
325 return;
326 }
327 }
328 throw new IncompatibleTypesException(p1, i, p2, j);
329 }
330 catch (ArrayIndexOutOfBoundsException e)
331 {
332 throw new IndexOutOfBoundsException(p1, i, p2, j);
333 }
334 }

342 public static Collection<Connection> getConnections(Processor p)
343 {
344 Set<Connection> connections = new HashSet<Connection>();
345 int p_id = p.getId();
346 for (int i = 0; i < p.getInputArity(); i++)
347 {
348 Pullable pl = p.getPullableInput(i);
349 if (pl == null)
350 {
351 continue;
352 }
353 Processor d_p = pl.getProcessor();
354 if (d_p == null)
355 {
356 continue;
357 }
358 connections.add(new Connection(d_p.getId(), pl.getPosition(), p_id, i));
359 }
360 for (int i = 0; i < p.getOutputArity(); i++)
361 {
362 Pushable ps = p.getPushableOutput(i);
363 if (ps == null)
364 {
365 continue;
366 }
367 Processor d_p = ps.getProcessor();
368 if (d_p == null)
369 {
370 continue;
371 }
372 connections.add(new Connection(p_id, i, d_p.getId(), ps.getPosition()));
373 }
374 return connections;
375 }
