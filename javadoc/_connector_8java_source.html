121  public static Processor connect(EventTracker tracker, Processor p1, int i, Processor p2, int j)
122  {
123  // First check for type compatibility
124  if (s_checkForTypes)
125  {
126  // This will throw an exception if the connection is impossible
127  checkForException(p1, i, p2, j);
128  }
129  if (p1 == p2)
130  {
131  // This is weird: you try to connect a processor to itself
132  throw new SelfLoopException(p1, i, p2, j);
133  }
134  // Pull
135  try
136  {
137  Pullable p1_out = p1.getPullableOutput(i);
138  p2.setPullableInput(j, p1_out);
139  }
140  catch (ArrayIndexOutOfBoundsException e)
141  {
142  throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
143  }
144  catch (UnsupportedOperationException e)
145  {
146  // It's OK. Some processors deliberately throw this
147  // exception to warn an end user that they don't have a pushable
148  // or a pullable, but the connector does not care.
149  }
150  // Push
151  try
152  {
153  Pushable p2_in = p2.getPushableInput(j);
154  p1.setPushableOutput(i, p2_in);
155  }
156  catch (ArrayIndexOutOfBoundsException e)
157  {
158  throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
159  }
160  catch (UnsupportedOperationException e)
161  {
162  // Same as above
163  }
164  if (tracker != null)
165  {
166  tracker.setConnection(p1.getId(), i, p2.getId(), j);
167  }
168  return p2;
169  }

217  public static Processor connect(EventTracker tracker, Processor ... procs)
218  {
219  if (procs.length == 1)
220  {
221  // If given only one processor, do nothing
222  return procs[0];
223  }
224  for (int j = 0; j < procs.length - 1; j++)
225  {
226  Processor p1 = procs[j];
227  Processor p2 = procs[j + 1];
228  int arity = p1.getOutputArity();
229  for (int i = 0; i < arity; i++)
230  {
231  connect(tracker, p1, i, p2, i);
232  }
233  }
234  return procs[procs.length - 1];

279  @SuppressWarnings("unused")
280  protected static void checkForException(Processor p1, int i, Processor p2, int j)
281  {
282  try
283  {
284  Class<?> out_class = p1.getOutputType(i);
285  if (s_checkForBounds && out_class == null)
286  {
287  // p1 has no output, so how would you connect it to p2?
288  throw new IndexOutOfBoundsException(p1, i, p2, j);
289  }
290  if (out_class != null && out_class.equals(Variant.class))
291  {
292  // Skip type checking
293  return;
294  }
295  /* @NotNull */ Set<Class<?>> in_classes = p2.getInputType(j);
296  if (in_classes.isEmpty())
297  {
298  if (s_checkForBounds)
299  {
300  // p2 has no input, so how would you connect it to p1?
301  throw new IndexOutOfBoundsException(p1, i, p2, j);
302  }
303  else
304  {
305  // We don't mind that we connect an output to something
306  // that has no input
307  return;
308  }
309  }
310  for (Class<?> in_class : in_classes)
311  {
312  if (in_class.equals(Variant.class) || in_class.isAssignableFrom(out_class)
313  || in_class.equals(Object.class))
314  {
315  // Found a compatible in/out pair of types: return without exception
316  return;
317  }
318  }
319  throw new IncompatibleTypesException(p1, i, p2, j);
320  }
321  catch (ArrayIndexOutOfBoundsException e)
322  {
323  throw new IndexOutOfBoundsException(p1, i, p2, j);
324  }
325  }
