124  public static Processor connect(EventTracker tracker, Processor p1, int i, Processor p2, int j)
125  {
126  // First check for type compatibility
127  if (s_checkForTypes)
128  {
129  // This will throw an exception if the connection is impossible
130  checkForException(p1, i, p2, j);
131  }
132  if (p1 == p2)
133  {
134  // This is weird: you try to connect a processor to itself
135  throw new SelfLoopException(p1, i, p2, j);
136  }
137  // Pull
138  try
139  {
140  Pullable p1_out = p1.getPullableOutput(i);
141  p2.setPullableInput(j, p1_out);
142  }
143  catch (ArrayIndexOutOfBoundsException e)
144  {
145  throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
146  }
147  catch (UnsupportedOperationException e)
148  {
149  // It's OK. Some processors deliberately throw this
150  // exception to warn an end user that they don't have a pushable
151  // or a pullable, but the connector does not care.
152  }
153  // Push
154  try
155  {
156  Pushable p2_in = p2.getPushableInput(j);
157  p1.setPushableOutput(i, p2_in);
158  }
159  catch (ArrayIndexOutOfBoundsException e)
160  {
161  throw new Connector.IndexOutOfBoundsException(p1, i, p2, j);
162  }
163  catch (UnsupportedOperationException e)
164  {
165  // Same as above
166  }
167  if (tracker != null)
168  {
169  tracker.setConnection(p1.getId(), i, p2.getId(), j);
170  }
171  return p2;
172  }

220  public static Processor connect(EventTracker tracker, Processor ... procs)
221  {
222  if (procs.length == 1)
223  {
224  // If given only one processor, do nothing
225  return procs[0];
226  }
227  for (int j = 0; j < procs.length - 1; j++)
228  {
229  Processor p1 = procs[j];
230  Processor p2 = procs[j + 1];
231  int arity = p1.getOutputArity();
232  for (int i = 0; i < arity; i++)
233  {
234  connect(tracker, p1, i, p2, i);
235  }
236  }
237  return procs[procs.length - 1];
238  }

282  @SuppressWarnings("unused")
283  protected static void checkForException(Processor p1, int i, Processor p2, int j)
284  {
285  try
286  {
287  Class<?> out_class = p1.getOutputType(i);
288  if (s_checkForBounds && out_class == null)
289  {
290  // p1 has no output, so how would you connect it to p2?
291  throw new IndexOutOfBoundsException(p1, i, p2, j);
292  }
293  if (out_class != null && out_class.equals(Variant.class))
294  {
295  // Skip type checking
296  return;
297  }
298  /* @NotNull */ Set<Class<?>> in_classes = p2.getInputType(j);
299  if (in_classes.isEmpty())
300  {
301  if (s_checkForBounds)
302  {
303  // p2 has no input, so how would you connect it to p1?
304  throw new IndexOutOfBoundsException(p1, i, p2, j);
305  }
306  else
307  {
308  // We don't mind that we connect an output to something
309  // that has no input
310  return;
311  }
312  }
313  for (Class<?> in_class : in_classes)
314  {
315  if (in_class.equals(Variant.class) || in_class.isAssignableFrom(out_class)
316  || in_class.equals(Object.class))
317  {
318  // Found a compatible in/out pair of types: return without exception
319  return;
320  }
321  }
322  throw new IncompatibleTypesException(p1, i, p2, j);
323  }
324  catch (ArrayIndexOutOfBoundsException e)
325  {
326  throw new IndexOutOfBoundsException(p1, i, p2, j);
327  }
328  }

336  public static Collection<Connection> getConnections(Processor p)
337  {
338  Set<Connection> connections = new HashSet<Connection>();
339  int p_id = p.getId();
340  for (int i = 0; i < p.getInputArity(); i++)
341  {
342  Pullable pl = p.getPullableInput(i);
343  if (pl == null)
344  {
345  continue;
346  }
347  Processor d_p = pl.getProcessor();
348  if (d_p == null)
349  {
350  continue;
351  }
352  connections.add(new Connection(d_p.getId(), pl.getPosition(), p_id, i));
353  }
354  for (int i = 0; i < p.getOutputArity(); i++)
355  {
356  Pushable ps = p.getPushableOutput(i);
357  if (ps == null)
358  {
359  continue;
360  }
361  Processor d_p = ps.getProcessor();
362  if (d_p == null)
363  {
364  continue;
365  }
366  connections.add(new Connection(p_id, i, d_p.getId(), ps.getPosition()));
367  }
368  return connections;
369  }
