454 @Override
455 protected void trackAssociations(Number x, Number y, Number z, EventTracker tracker)
456 {
457 if (x.floatValue() == 0)
458 {
459 tracker.associateToOutput(-1, 0, 0, 0, 0);
460 }
461 else if (y.floatValue() == 0)
462 {
463 tracker.associateToOutput(-1, 1, 0, 0, 0);
464 }

472 @Override
473 public boolean evaluatePartial(Object[] inputs, Object[] outputs, Context context)
474 {
475 if (inputs[0] != null && ((Number) inputs[0]).floatValue() == 0f)
476 {
477 outputs[0] = 0f;
478 return true;
479 }
480 if (inputs[1] != null && ((Number) inputs[1]).floatValue() == 0f)
481 {
482 outputs[0] = 0f;
483 return true;
484 }
485 if (inputs[0] != null && inputs[1] != null)
486 {
487 outputs[0] = ((Number) inputs[0]).floatValue() * ((Number) inputs[1]).floatValue();
488 return true;
489 }
490 return false;
491 }

539 public static final Number getNumber(Object x)
540 {
541 if (x instanceof Number)
542 {
543 return (Number) x;
544 }
545 if (!(x instanceof String))
546 {
547 // Anything but a string: work on the value of toString
548 x = x.toString();
549 }
550 if (x instanceof String)
551 {
552 try
553 {
554 return Integer.parseInt((String) x);
555 }
556 catch (NumberFormatException e)
557 {
558 try
559 {
560 return Float.parseFloat((String) x);
561 }
562 catch (NumberFormatException e2)
563 {
564 throw new FunctionException(e2);
565 }
566 }
567 }
568 throw new FunctionException("Object incompatible with Number");
569 }
