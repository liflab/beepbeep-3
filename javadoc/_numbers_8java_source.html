444  public static final Number getNumber(Object x)
445  {
446  if (x instanceof Number)
447  {
448  return (Number) x;
449  }
450  if (!(x instanceof String))
451  {
452  // Anything but a string: work on the value of toString
453  x = x.toString();
454  }
455  if (x instanceof String)
456  {
457  try
458  {
459  return Integer.parseInt((String) x);
460  }
461  catch (NumberFormatException e)
462  {
463  try
464  {
465  return Float.parseFloat((String) x);
466  }
467  catch (NumberFormatException e2)
468  {
469  throw new FunctionException(e2);
470  }
471  }
472  }
473  throw new FunctionException("Object incompatible with Number");
474  }
