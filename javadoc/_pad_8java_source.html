52 public Pad(Processor p, int times, Object ... front)
53 {
54 super(p.getInputArity(), p.getOutputArity());
55 m_processor = p;
56 m_times = times;
57 m_front = front;
58 m_numPad = 0;
59 m_innerPushables = new Pushable[p.getInputArity()];
60 for (int i = 0; i < m_innerPushables.length; i++)
61 {
62 m_innerPushables[i] = m_processor.getPushableInput(i);
63 }
64 m_sink = new QueueSink(m_processor.getOutputArity());
65 Connector.connect(m_processor, m_sink);
66 }
67
68 @Override
69 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
70 {
71 for (int i = 0; i < inputs.length; i++)
72 {
73 m_innerPushables[i].push(inputs[i]);
74 }
75 if (m_numPad < m_times)
76 {
77 m_numPad++;
78 outputs.add(m_front);
79 return true;
80 }
81 boolean loop = true;
82 while (loop)
83 {
84 for (int i = 0; i < m_sink.getInputArity(); i++)
85 {
86 if (m_sink.getQueue(i).isEmpty())
87 {
88 loop = false;
89 break;
90 }
91 }
92 if (loop)
93 {
94 Object[] out = new Object[m_sink.getInputArity()];
95 for (int i = 0; i < m_sink.getInputArity(); i++)
96 {
97 out[i] = m_sink.getQueue(i).remove();
98 }
99 outputs.add(out);
100 }
101 }
102 return true;
103 }

106 public void reset()
107 {
108 m_numPad = 0;
109 m_processor.reset();
110 }
