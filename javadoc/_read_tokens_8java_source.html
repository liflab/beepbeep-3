113 protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
114 {
115 Queue<Object[]> q = new ArrayDeque<Object[]>(1);
116 boolean b = super.compute(inputs, q);
117 if (!b && !m_lastChunk.isEmpty())
118 {
119 prepareChunk(m_lastChunk, outputs);
120 m_lastChunk = "";
121 return false;
122 }
123 if (!q.isEmpty())
124 {
125 String s = (String) ((Object[]) q.poll())[0];
126 m_lastChunk += s;
127 if (m_lastChunk.contains(m_separator))
128 {
129 boolean dangling = !m_lastChunk.endsWith(m_separator);
130 String[] parts = s.split(m_separator);
131 for (int i = 0; i < parts.length - (dangling ? 1 : 0); i++)
132 {
133 prepareChunk(parts[i], outputs);
134 }
135 if (dangling)
136 {
137 m_lastChunk = parts[parts.length - 1];
138 }
139 else
140 {
141 m_lastChunk = "";
142 }
143 }
144 }
145 return b;
146 }
