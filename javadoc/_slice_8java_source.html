131  @Override
132  protected boolean compute(Object[] inputs, Object[] outputs)
133  {
134  int output_arity = getOutputArity();
135  Object[] f_value = new Object[1];
136  try
137  {
138  m_slicingFunction.evaluate(inputs, f_value);
139  }
140  catch (FunctionException e)
141  {
142  throw new ProcessorException(e);
143  }
144  Object slice_ids = f_value[0];
145  if (slice_ids == null)
146  {
147  // This event applies to no slice; don't bother processing it
148  outputs[0] = m_lastValues;
149  return true;
150  }
151  Object[] slice_vals;
152  if (m_explodeArrays)
153  {
154  if (slice_ids.getClass().isArray())
155  {
156  slice_vals = (Object[]) slice_ids;
157  }
158  if (slice_ids instanceof Collection)
159  {
160  Collection<?> col = (Collection<?>) slice_ids;
161  slice_vals = new Object[col.size()];
162  int i = 0;
163  for (Object o : col)
164  {
165  slice_vals[i] = o;
166  i++;
167  }
168  }
169  else
170  {
171  slice_vals = new Object[] { slice_ids };
172  }
173  }
174  else
175  {
176  slice_vals = new Object[] { slice_ids };
177  }
178  for (Object slice_id : slice_vals)
179  {
180  Set<Object> slices_to_process = new HashSet<Object>();
181  if (slice_id instanceof ToAllSlices || slice_id == null)
182  {
183  slices_to_process.addAll(m_slices.keySet());
184  }
185  else
186  {
187  if (!m_slices.containsKey(slice_id))
188  {
189  // First time we see this value: create new slice
190  Processor p = m_processor.duplicate();
191  m_slices.put(slice_id, p);
192  addContextFromSlice(p, slice_id);
193  QueueSink sink = new QueueSink(output_arity);
194  Connector.connect(p, sink);
195  m_sinks.put(slice_id, sink);
196  // Put dummy value temporarily
197  m_lastValues.put(slice_id, null);
198  }
199  slices_to_process.add(slice_id);
200  }
201  @SuppressWarnings("unchecked")
202  Future<Pushable>[] futures = new Future[slices_to_process.size()];
203  for (Object s_id : slices_to_process)
204  {
205  // Find processor corresponding to that slice
206  Processor slice_p = m_slices.get(s_id);
207  // If this slice hasn't been cleaned up...
208  if (slice_p != null)
209  {
210  QueueSink sink_p = m_sinks.get(s_id);
211  // Push the input into the processor
212  // Pushable[] p_array = new Pushable[inputs.length];
213  for (int i = 0; i < inputs.length; i++)
214  {
215  Object o_i = inputs[i];
216  Pushable p = slice_p.getPushableInput(i);
217  Future<Pushable> f = p.pushFast(o_i);
218  if (f == null)
219  {
220  System.out.println("NULL");
221  }
222  futures[i] = f;
223  // p_array[i] = p;
224  }
225  // Wait for all slices to be done
226  for (Future<Pushable> f : futures)
227  {
228  if (f != null)
229  {
230  try
231  {
232  f.get();
233  }
234  catch (InterruptedException e)
235  {
236  throw new ProcessorException(e);
237  }
238  catch (ExecutionException e)
239  {
240  throw new ProcessorException(e);
241  }
242  }
243  }
244  // Collect the output from that processor
245  Object[] out = sink_p.remove();
246  // Can we clean that slice?
247  Object[] can_clean = new Object[1];
248  if (m_cleaningFunction != null)
249  {
250  try
251  {
252  m_cleaningFunction.evaluate(out, can_clean);
253  }
254  catch (FunctionException e)
255  {
256  throw new ProcessorException(e);
257  }
258  }
259  if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
260  && (Boolean) (can_clean[0]))
261  {
262  // Yes: remove the processor for that slice
263  m_slices.remove(s_id);
264  m_sinks.remove(s_id);
265  }
266  m_lastValues.put(s_id, out[0]);
267  }
268  }
269  outputs[0] = m_lastValues;
270  }
271  return true;
272  }
