98 @Override
99 public Slice readState(Object o) throws ProcessorException
100 {
101 Map<String,Object> contents = (HashMap<String,Object>) o;
102 Function cleaning_function = (Function) contents.get("cleaning-function");
103 Function slicing_function = (Function) contents.get("slicing-function");
104 Processor proc = (Processor) contents.get("processor");
105 Slice s = new Slice(slicing_function, proc, cleaning_function);
106 s.m_lastValues = (HashMap<Object,Object>) contents.get("last-values");
107 s.m_sinks = (HashMap<Object,QueueSink>) contents.get("sinks");
108 s.m_slices = (HashMap<Object,Processor>) contents.get("slices");
109 s.m_explodeArrays = (Boolean) contents.get("explode-arrays");
110 // Connect slices to sinks
111 for (Map.Entry<Object,Processor> entry : s.m_slices.entrySet())
112 {
113 Processor p = entry.getValue();
114 if (!s.m_sinks.containsKey(entry.getKey()))
115 {
116 throw new ProcessorException("Cannot restore the state of the slice processor");
117 }
118 QueueSink qs = s.m_sinks.get(entry.getKey());
119 Connector.connect(p, qs);
120 }
121 return s;
122 }

124 @Override
125 public Slice duplicate(boolean with_state)
126 {
127 Slice s = null;
128 if (m_cleaningFunction == null)
129 {
130 s = new Slice(m_slicingFunction.duplicate(with_state), m_processor.duplicate(with_state));
131 }
132 else
133 {
134 s = new Slice(m_slicingFunction.duplicate(with_state), m_processor.duplicate(with_state),
135 m_cleaningFunction.duplicate(with_state));
136 }
137 super.copyInto(s, with_state);
138 if (with_state)
139 {
140 s.m_lastValues.putAll(m_lastValues);
141 }
142 return s;
143 }
